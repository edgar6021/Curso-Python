while True:
    try:
        ping3.ping("10.0.0.73")
        print("exito")

        print("La respuesta del host es muy lenta")
    except ping3.errors.PingError:  # si el host no responde f `PingError`.
        print("Host inaccesible")

    time.sleep(5)
